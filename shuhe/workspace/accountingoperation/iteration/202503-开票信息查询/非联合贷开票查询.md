-- 订单客账信息
with t1 as
(
select b.uid
     , d.cus_nam
     , b.bill_no as order_no
     , b.term_no
     , b.term_id
     , c.loan_asset_id
     , b.paid_prin/100.00 as paid_prin
     , b.paid_int/100.00 as paid_int
     , b.paid_guarantee_fee/100.00 as paid_guarantee_fee
     , b.paid_amc_fee/100.00 as paid_amc_fee
     , b.paid_late_fee/100.00 as paid_late_fee
     , b.paid_pen/100.00 as paid_pen
     , b.paid_pre_fee/100.00 as paid_pre_fee
     , b.paid_amt/100.00 as paid_amt
     , b.paid_out_date
from ods_pdw_loan.ods_pdw_loan_dsloankernelali_loancore_cls_term_df b
left join dwt.dwt_heavy_order_df c
  on c.ds = '${bizdate}'
 and c.loan_success_flag = '1'
 and b.bill_no = c.order_no
join dbus.dbus_user_apply_info_df d
  on d.ds = '${bizdate}'
 and b.uid = d.uid
where  b.ds = '${bizdate}'
 and b.paid_amt > 0
),
-- 关联补偿退款金额

t2 as
(
select  a.uid
     , a.cus_nam
     , a.order_no
     , a.term_no
     , a.term_id
     , a.loan_asset_id
     , a.paid_prin
     , a.paid_int
     , a.paid_guarantee_fee
     , a.paid_amc_fee
     , a.paid_late_fee
     , a.paid_pen
     , a.paid_pre_fee
     , a.paid_amt
     , a.paid_out_date
     , coalesce(b.refund_amt,0) as refund_amt
     , c.whether_amc
from t1 a 
left join 
         (
          select coalesce(a.bill_no,b.bill_no) as order_no
               , coalesce(a.refund_amt,0) + coalesce(b.rebate_amount,0) as refund_amt
          from
              (
               select bill_no , sum(coalesce(sch_refund_amt,0) + coalesce(tot_compensate_refund,0))/100.00 as refund_amt
               -- 结清返现金额由tot_refund_amt 累计实退金额 改为 sch_refund_amt 累计应退款金额
               from ods_pdw_loan.ods_pdw_loan_dsloankernelali_loancore_cls_loan_df
               where ds = '${bizdate}'  
               group by bill_no
              ) a
          full join
                   (
                    select  bill_no  , sum(rebated_amt)/100.00 as rebate_amount
                    from ods_pdw_loan.ods_pdw_loan_dsloankernelali_loancore_cls_term_ext_df
                    where ds = '${bizdate}'  
                    group by bill_no
                   ) b
                on a.bill_no = b.bill_no
            
         ) b
      on a.order_no = b.order_no
left join ods_pdw_loan.ods_pdw_loan_dsloankernelali_loancore_cls_loan_df c 
       on c.ds = '${bizdate}'
      and a.order_no = c.bill_no
),
-- 关联资方信息
t3 as
(
SELECT  a.uid, a.cus_nam, a.order_no, a.term_no
            , a.term_id, COALESCE(d.asset_id, a.loan_asset_id) AS loan_asset_id, a.paid_prin
            , a.paid_int, a.paid_guarantee_fee, a.paid_amc_fee, a.paid_late_fee, a.paid_pen
            , a.paid_pre_fee, a.paid_amt, a.paid_out_date, a.refund_amt
            , CASE 
                WHEN d.stage_order_no IS NULL THEN COALESCE(b.act_rep_tot_amt / 100.00, c.actual_repaid_total_amount)
                WHEN d.stage_order_no IS NOT NULL THEN d.actual_repaid_total_amount
            END AS act_rep_tot_amt
            , CASE 
                WHEN d.stage_order_no IS NULL THEN COALESCE(b.act_rep_pri / 100.00, c.schedule_principal)
                WHEN d.stage_order_no IS NOT NULL THEN d.actual_repaid_principal
            END AS act_rep_pri, a.whether_amc
        FROM t2 a
        LEFT  JOIN
        (select stg_ord_no
               ,stg_no
               ,act_rep_tot_amt
               ,act_rep_pri
           from
              (
               select stg_ord_no
                    ,stg_no
                    ,act_rep_tot_amt
                    ,act_rep_pri
                    ,row_number() over(partition by stg_ord_no,stg_no order by act_rep_tot_amt desc) as rn
               from ads_app_fundcore.ads_app_fundcore_repay_plan_process_nrt_hvy
               where ds='${bizdate}'
              ) t
           where rn = 1
         )

        b  -- 已去重
        ON  a.order_no = b.stg_ord_no
            AND a.term_no = b.stg_no
        LEFT OUTER JOIN cdmx.cdmx_fct_heavy_fund_stage_plan_df c
        ON c.ds = '${bizdate}'
            AND a.order_no = c.order_no
            AND a.term_no = c.stage_no
            AND c.fund_code IS NOT NULL
left join 
         (
          /*
          1.双池信托订单在资方分期计划表里每笔分期会记两次（放款池和承接池），金额也会记两次
          2.actual_label只会在实际结算的资方记录，债转前记录在放款池，债转后记录在承接池
          3.pdm_acctops.pdm_acctops_transfer_stage_plan_basic_df为涉及债转资方的分期计划表
          */
          select a.stage_order_no
               , a.stage_no
               , case when b.is_transfer_plan = '1' then und_asset_id
                      when b.is_transfer_plan = '0' then ori_asset_id
                 end as asset_id
               , date(a.fund_repaid_time) as fund_repaid_time
               , sum(a.actual_repaid_principal)/100.00 as actual_repaid_principal
               , sum(a.actual_repaid_total_amount)/100.00 as actual_repaid_total_amount
          from ods_pdw_loan.ods_pdw_loan_dsfundwallali_capitalaccount_repay_plan_df a 
          join pdm_acctops.pdm_acctops_transfer_stage_plan_basic_df b
            on b.ds = '${bizdate}'
           and a.stage_order_no = b.stage_order_no
           and a.stage_no = b.stage_no
          where a.ds = '${bizdate}'   and  a.stage_order_no='CH202509011346218'
            and a.actual_label in ('compensatory','buyback','benefit')
          group by a.stage_order_no
               , a.stage_no
               , case when b.is_transfer_plan = '1' then und_asset_id
                      when b.is_transfer_plan = '0' then ori_asset_id
                 end
               , date(a.fund_repaid_time)
         ) d
      on a.order_no = d.stage_order_no
     and a.term_no = d.stage_no
),
t7 as
(
select
 a.uid
     , a.cus_nam
     , a.order_no
     , a.term_no
     , a.term_id
     , a.loan_asset_id
     , c.asset_name
     , a.paid_prin
     , a.paid_int
     , a.paid_guarantee_fee
     , a.paid_amc_fee
     , a.paid_late_fee
     , a.paid_pen
     , a.paid_pre_fee
     , a.paid_amt
     , a.paid_out_date
     , a.refund_amt
     , a.act_rep_tot_amt
     , a.act_rep_pri
     , c.fund_name
     , case when b.main_bill_no is not null then 1
            else 0 
       end as is_lhd
     , a.whether_amc
from t3 a
left join
         (
          select main_bill_no
          from ods_pdw_loan.ods_pdw_loan_dsloankernel_loancore_cls_sub_loan_df
          where ds = '${bizdate}'
          group by main_bill_no
         ) b
      on a.order_no = b.main_bill_no
left join pdm_fund.pdm_fund_zjpz_basicconfig_new_di c
       on c.ds = '${bizdate}'
      and a.loan_asset_id = c.asset_id
),
-- 标准助贷：关联担保信息
t4 as
(
select 
 a.uid
     , a.cus_nam
     , a.order_no
     , a.term_no
     , a.term_id
     , a.loan_asset_id
     , a.paid_prin
     , a.paid_int
     , a.paid_guarantee_fee
     , a.paid_amc_fee
     , a.paid_late_fee
     , a.paid_pen
     , a.paid_pre_fee
     , a.paid_amt
     , a.paid_out_date
     , a.refund_amt
     , a.act_rep_tot_amt
     , a.act_rep_pri
     , b.guarantee_model
     , case when a.loan_asset_id not like '%FOCUS_LOAN%' and b.guarantee_model is null and c.guarantee = 'flase' then '无担保'
            when a.loan_asset_id not like '%FOCUS_LOAN%' and b.guarantee_model is null and c.guarantee = 'true' then c.guaranteecompany
            when a.loan_asset_id not like '%FOCUS_LOAN%' and b.guarantee_model is null then ''
            else b.guarantee_company
       end as guarantee_company
     , b.guarantee_rate
     , b.follow_guarantee_company
     , b.follow_guarantee_rate
     , case when a.loan_asset_id like '%FOCUS_LOAN%' and a.loan_asset_id not like '%ZHX%' then '小贷'
            when a.loan_asset_id like '%FOCUS_LOAN%' and a.loan_asset_id like '%ZHX%' then '小贷+担保'
            when b.guarantee_model = '独立' or b.guarantee_model is null then '非小贷（独立担保）'
            when b.guarantee_model = '联合' then '非小贷（联合担保）'
            else '其他'
       end as bank_type
     , c.fund_name
     , c.asset_name
from t7 a
left join 
         (
          select a.stage_order_no
               , case when b.stage_order_no is not null then '联合'
                      when b.stage_order_no is null then '独立'
                 end as guarantee_model
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then b.guarantee_company
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then a.guarantee_company
                      else a.guarantee_company
                 end as guarantee_company
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then b.guarantee_rate
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then a.guarantee_rate
                      else a.guarantee_rate
                 end as guarantee_rate
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then a.guarantee_company
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then b.guarantee_company
                      else b.guarantee_company
                 end as follow_guarantee_company
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then a.guarantee_rate
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then b.guarantee_rate
                      else b.guarantee_rate
                 end as follow_guarantee_rate
          from ods_pdw_loan.ods_pdw_loan_dsguaranteecore_guaranteecore_guarantee_stage_order_df a
          left join ods_pdw_loan.ods_pdw_loan_dsguaranteecore_guaranteecore_guarantee_stage_order_df b
                 on b.ds = '${bizdate}'
                and a.stage_order_no = b.stage_order_no
                and b.main_guarantee_company = 0
          where a.ds = '${bizdate}'
            and a.main_guarantee_company = 1
          group by a.stage_order_no
               , case when b.stage_order_no is not null then '联合'
                      when b.stage_order_no is null then '独立'
                 end
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then b.guarantee_company
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then a.guarantee_company
                      else a.guarantee_company
                 end
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then b.guarantee_rate
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then a.guarantee_rate
                      else a.guarantee_rate
                 end
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then a.guarantee_company
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then b.guarantee_company
                      else b.guarantee_company
                 end
               , case when b.stage_order_no is not null and a.guarantee_company like '%中禾信%' then a.guarantee_rate
                      when b.stage_order_no is not null and b.guarantee_company like '%中禾信%' then b.guarantee_rate
                      else b.guarantee_rate
                 end
         ) b
      on a.order_no = b.stage_order_no
left join pdm_fund.pdm_fund_zjpz_basicconfig_new_di c
       on c.ds = '${bizdate}'
      and a.loan_asset_id = c.asset_id
where a.is_lhd = 0
  and coalesce(a.whether_amc,'N') <> 'Y' 
),
-- 标准助贷
t5 as 
(
select 
 a.uid
     , a.cus_nam
     , a.order_no
     , a.term_no
     , a.term_id
     , a.loan_asset_id
     , a.asset_name
     , round(a.paid_prin,2) as paid_prin
     , round(a.paid_int,2) as paid_int
     , round(a.paid_guarantee_fee,2) as paid_guarantee_fee
     , round(a.paid_amc_fee,2) as paid_amc_fee
     , round(a.paid_late_fee,2) as paid_late_fee
     , round(a.paid_pen,2) as paid_pen
     , round(a.paid_pre_fee,2) as paid_pre_fee
     , round(a.paid_amt,2) as paid_amt
     , a.paid_out_date
     , round(a.refund_amt,2) as refund_amt
     , round(a.act_rep_tot_amt,2) as act_rep_tot_amt
     , round(a.act_rep_pri,2) as act_rep_pri
     , a.guarantee_rate/100 as guarantee_rate
     , case when a.loan_asset_id like '%FOCUS_LOAN%' then '重庆市分众小额贷款有限公司'
            else a.fund_name
       end as fund_name
     , case when a.bank_type = '小贷' then round(a.paid_amt - a.paid_prin - a.refund_amt,2)
            else round(a.act_rep_tot_amt - a.act_rep_pri,2)
       end as bank_amt
     , case when a.bank_type = '小贷+担保' then '中禾信融资担保(福建)有限公司' 
            when a.bank_type = '非小贷（独立担保）' then a.guarantee_company
       end as ind_gua_owner
     , case when a.bank_type in ('小贷+担保','非小贷（独立担保）') then round(a.paid_amt - a.refund_amt - a.act_rep_tot_amt,2)
       end as ind_gua_amt
     , case when a.bank_type = '非小贷（联合担保）' then a.guarantee_company
       end as uni_gua_main_owner
     , case when a.bank_type = '非小贷（联合担保）' then round(b.paid_guarantee_fee/100.00,2)
       end as uni_gua_main_amt
     , case when a.bank_type = '非小贷（联合担保）' then a.follow_guarantee_company
       end as uni_gua_follow_owner
     , case when a.bank_type = '非小贷（联合担保）' then round(a.paid_amt - a.refund_amt - a.act_rep_tot_amt - b.paid_guarantee_fee/100.00,2) 
       end as uni_gua_follow_amt
from t4 a 
left join ods_pdw_loan.ods_pdw_loan_dsguaranteecore_guaranteecore_guarantee_stage_plan_df b
       on b.ds = '${bizdate}'
      and a.order_no = b.stage_order_no
      and a.term_no = b.stage_no
      and substring_index(b.guarantee_order_no,'_',-1) = '0'
),
-- amc订单
t6 as
(
select 
 a.uid
     , a.cus_nam
     , a.order_no
     , a.term_no
     , a.term_id
     , a.loan_asset_id
     , a.asset_name
     , round(a.paid_prin,2) as paid_prin
     , round(a.paid_int,2) as paid_int
     , round(a.paid_guarantee_fee,2) as paid_guarantee_fee
     , round(a.paid_amc_fee,2) as paid_amc_fee
     , round(a.paid_late_fee,2) as paid_late_fee
     , round(a.paid_pen,2) as paid_pen
     , round(a.paid_pre_fee,2) as paid_pre_fee
     , round(a.paid_amt,2) as paid_amt
     , a.paid_out_date
     , round(a.refund_amt,2) as refund_amt
     , round(a.act_rep_tot_amt,2) as act_rep_tot_amt
     , round(a.act_rep_pri,2) as act_rep_pri
     , c.guarantee_rate/100 as guarantee_rate
     , b.amc_company as amc_name
     , round(coalesce(a.paid_amc_fee,0) + coalesce(a.paid_pre_fee,0) + coalesce(a.paid_pen,0),2) as amc_amt
     , a.fund_name
     , round(a.act_rep_tot_amt - a.act_rep_pri,2) as bank_amt
     , c.guarantee_company as ind_gua_owner
     , round(coalesce(a.paid_amt,0) - coalesce(a.act_rep_tot_amt,0) - coalesce(a.paid_amc_fee,0) - coalesce(a.paid_pre_fee,0) - coalesce(a.paid_pen,0) - coalesce(a.refund_amt,0),2) as ind_gua_amt
from t7 a
left join 
         (
          select a.bill_no
               , b.amc_company
          from ods_pdw_loan.ods_pdw_loan_dsloankernelali_loancore_cls_loan_df a
          left join ods_pdw_loan.ods_pdw_loan_dsfundwall_fundwall_amc_company_df b
                 on a.amc_company = b.amc_code
                and b.ds = '${bizdate}'
          where a.ds = '${bizdate}'
            and a.whether_amc = 'Y'
          group by a.bill_no
               , b.amc_company
         ) b
      on a.order_no = b.bill_no
left join ods_pdw_loan.ods_pdw_loan_dsguaranteecore_guaranteecore_guarantee_stage_order_df c
       on c.ds = '${bizdate}'
      and c.main_guarantee_company = 1
      and a.order_no = c.stage_order_no
where a.is_lhd = 0
  and a.whether_amc = 'Y' 
)
